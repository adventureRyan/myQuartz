### 打开摄像头
```python
import cv2
import numpy as np
def video_demo():
    capture = cv2.VideoCapture(0)#0为电脑内置摄像头
    while(True):
        ret, frame = capture.read()#摄像头读取,ret为是否成功打开摄像头,true,false。 frame为视频的每一帧图像
        frame = cv2.flip(frame, 1)#摄像头是和人对立的，将图像左右调换回来正常显示。
        cv2.imshow("video", frame)
        c = cv2.waitKey(50)
        if c == 27:
            break
video_demo()
cv2.destroyAllWindows()

```

### 打开摄像头并截屏

```python
import cv2

cap = cv2.VideoCapture(0)  # 打开摄像头

while (1):
    # get a frame
    ret, frame = cap.read()
    frame = cv2.flip(frame, 1)  # 摄像头是和人对立的，将图像左右调换回来正常显示

    cv2.imshow("capture", frame)  # 生成摄像头窗口

        cv2.imwrite("test.png", frame)  # 保存路径
        break

cap.release()
cv2.destroyAllWindows()
```

## 电脑端成功通信
要使用 Python 的 OpenCV 和 Socket 库实现实时摄像头数据共享，你可以遵循以下步骤。这种实现方式包括两个部分：一个是服务器端，它从摄像头读取视频流并通过 Socket 发送；另一个是客户端，它通过 Socket 接收视频流并显示出来。

### 1. 安装依赖
你需要确保安装了以下 Python 库：
- OpenCV (`cv2`)
- Socket (Python 自带的标准库)

如果没有安装 OpenCV，你可以通过以下命令安装：
```bash
pip install opencv-python
```

### 2. 服务器端代码
服务器端代码负责从摄像头捕获视频帧，并通过 Socket 将其发送到客户端。

```python
import cv2
import socket
import pickle
import struct

# 创建Socket对象
server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
host_name = socket.gethostname()
host_ip = "192.168.1.151"
port = 8898

# 绑定Socket到指定IP和端口
server_socket.bind((host_ip, port))
server_socket.listen(5)
print(f"Listening at {host_ip}:{port}")

# 接受客户端连接
client_socket, addr = server_socket.accept()
print('Got connection from:', addr)

# 使用OpenCV捕获视频
vid = cv2.VideoCapture(0)

while vid.isOpened():
    ret, frame = vid.read()
    if not ret:
        break

    # 序列化帧数据
    data = pickle.dumps(frame)
    # 打包序列化数据长度
    message = struct.pack("Q", len(data)) + data

    # 发送数据给客户端
    client_socket.sendall(message)

    # 显示帧（在服务器端）
    cv2.imshow('Sending...', frame)

    # 按下 'q' 键退出
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# 释放资源
vid.release()
client_socket.close()
server_socket.close()
```

### 3. 客户端代码
客户端代码负责通过 Socket 接收视频帧，并使用 OpenCV 显示。

```python
import cv2
import socket
import pickle
import struct

# 创建Socket对象
client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
host_ip = "192.168.1.151"  # 替换为服务器的IP地址
port = 8898

# 连接到服务器
client_socket.connect((host_ip, port))

data = b""
payload_size = struct.calcsize("Q")

while True:
    # 循环获取数据
    while len(data) < payload_size:
        packet = client_socket.recv(4*1024)  # 4K
        if not packet:
            break
        data += packet

    packed_msg_size = data[:payload_size]
    data = data[payload_size:]
    msg_size = struct.unpack("Q", packed_msg_size)[0]

    while len(data) < msg_size:
        data += client_socket.recv(4*1024)

    frame_data = data[:msg_size]
    data = data[msg_size:]

    # 反序列化数据
    frame = pickle.loads(frame_data)

    # 显示帧
    cv2.imshow("Receiving...", frame)

    # 按下 'q' 键退出
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

client_socket.close()
cv2.destroyAllWindows()
```

### 4. 运行步骤
1. 先在服务器端运行服务器脚本。
2. 确保客户端和服务器在同一网络环境下。
3. 在客户端运行客户端脚本，将 `host_ip` 替换为服务器的 IP 地址。

> Attention
> python 版本不能低于 3.8,不然会报错出现 `No module named 'numpy._core`
